```typescript
export type Column<T> = {
  key: string
  header: string
  render: (row: T) => React.ReactNode
  primary?: boolean
  hideInCard?: boolean
  align?: "left" | "right"
  cardLabel?: string
}

export type ResponsiveTableProps<T> = {
  columns: Column<T>[]
  rows: T[]
  getRowKey: (row: T) => string
  getRowProps?: (row: T) => React.HTMLAttributes<HTMLElement>
  emptyState?: React.ReactNode
}

export function ResponsiveTable<T>(props: ResponsiveTableProps<T>): React.ReactElement | null
```
