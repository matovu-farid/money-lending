2. Create `src/lib/auth.ts` -- Better Auth server instance with first-user-superadmin hook:
   ```typescript
   import { betterAuth } from "better-auth"
   import { drizzleAdapter } from "better-auth/adapters/drizzle"
   import { admin } from "better-auth/plugins"
   import { db } from "./db"
   import { ac, superAdminRole, adminRole, loanOfficerRole, unassignedRole } from "./permissions"

   export const auth = betterAuth({
     database: drizzleAdapter(db, { provider: "pg" }),
     emailAndPassword: { enabled: true },
     session: {
       cookieCache: {
         enabled: true,
         maxAge: 5 * 60, // 5 minutes
       },
     },
     databaseHooks: {
       user: {
         create: {
           after: async (user) => {
             // FIRST-USER BOOTSTRAP: If this is the very first user in the
             // database, promote them to superAdmin so the role hierarchy
             // can be bootstrapped without manual DB intervention.
             // The `user` table is managed by Better Auth. Use Drizzle's
             // raw `db` to count rows. At the point this hook fires the
             // newly created user already exists in the table, so count === 1
             // means this is the first user.
             //
             // Import the user table from the generated auth schema.
             // NOTE: The generated schema file must exist before this runs
             // (it is generated in step 4 below and the import is resolved
             // at build time). Use a dynamic import or direct SQL count to
             // avoid circular dependency issues if the schema barrel
             // re-exports auth tables:
             const { sql } = await import("drizzle-orm")
             const result = await db.execute(sql`SELECT count(*)::int AS cnt FROM "user"`)
             const count = result.rows?.[0]?.cnt ?? result[0]?.cnt ?? 0

             if (Number(count) === 1) {
               // This is the first user -- promote to superAdmin.
               // Use Better Auth's internal API to set the role on the
               // user record. We update directly via Drizzle since we are
               // inside a databaseHook and calling auth.api.setRole here
               // could cause recursion or require request headers we don't have.
               await db.execute(
                 sql`UPDATE "user" SET "role" = 'superAdmin' WHERE "id" = ${user.id}`
               )
             }
           },
         },
       },
     },
     plugins: [
       admin({
         ac,
         roles: {
           superAdmin: superAdminRole,
           admin: adminRole,
           loanOfficer: loanOfficerRole,
           unassigned: unassignedRole,
         },
         defaultRole: "unassigned",
       }),
     ],
   })
   ```

   IMPORTANT NOTES on the first-user hook:
   - Uses `databaseHooks.user.create.after` -- this fires after the user row is inserted by Better Auth.
   - At hook execution time, the new user is already in the `user` table, so `count === 1` means this is the first-ever user.
   - Uses raw SQL via Drizzle (`db.execute(sql...)`) to both count users and update the role. This avoids importing the generated auth schema (which may cause circular deps) and avoids calling `auth.api.setRole` inside a hook (no request headers available).
   - The `"user"` table name is Better Auth's default table name. If the generated schema uses a different name, adjust accordingly.
   - All subsequent users still get `defaultRole: "unassigned"` -- only the very first user is promoted.

1. Create `src/proxy.ts` (NOT middleware.ts -- Next.js 16 breaking change).

   CRITICAL: This proxy must call `auth.api.getSession` for FULL session validation with role checking -- NOT a cookie-only check. The cookie-only approach was considered in RESEARCH.md Open Question #1, but it fails to redirect "unassigned" users to `/pending-approval` because a cookie check cannot determine the user's role. The proxy MUST inspect `session.user.role` to enforce AUTH-01 (unassigned users see /pending-approval, not the app).

   Implementation matching RESEARCH.md Pattern 2:
   ```typescript
   import { NextResponse } from "next/server"
   import type { NextRequest } from "next/server"
   import { auth } from "@/lib/auth"

   const AUTH_PAGES = ["/login", "/register", "/forgot-password"]

   export async function proxy(request: NextRequest) {
     const session = await auth.api.getSession({
       headers: request.headers,
     })

     const { pathname } = request.nextUrl
     const isAuthPage = AUTH_PAGES.some((p) => pathname.startsWith(p))

     // No valid session -- redirect to login unless on auth page
     if (!session?.user) {
       if (isAuthPage) return NextResponse.next()
       return NextResponse.redirect(new URL("/login", request.url))
     }

     // Unassigned users can ONLY access /pending-approval -- redirect everywhere else
     if (session.user.role === "unassigned") {
       if (pathname === "/pending-approval") return NextResponse.next()
       return NextResponse.redirect(new URL("/pending-approval", request.url))
     }

     // Authenticated + assigned user visiting auth pages or /pending-approval -- redirect to dashboard
     if (isAuthPage || pathname === "/pending-approval") {
       return NextResponse.redirect(new URL("/dashboard", request.url))
     }

     // Authenticated user with an assigned role -- allow through
     return NextResponse.next()
   }

   export const config = {
     matcher: ["/((?!api|_next/static|_next/image|favicon.ico).*)"],
   }
   ```

   KEY DIFFERENCES from the previous cookie-only approach:
   - Uses `auth.api.getSession({ headers: request.headers })` instead of reading cookie directly
   - Checks `session.user.role === "unassigned"` and redirects to `/pending-approval`
   - Assigned users visiting `/pending-approval` are redirected to `/dashboard` (prevents stale page)
   - Better Auth's `cookieCache` (configured in Task 1 with 5-min maxAge) mitigates the DB lookup cost

   NOTE on first-user flow: The first registered user gets superAdmin role (via the databaseHook in Task 1), so they will NOT have `role === "unassigned"` and will NOT be redirected to /pending-approval. They go straight to /dashboard. This is correct -- no special proxy logic needed for the first user.

2. Create `src/actions/user.actions.ts` -- Server Action for role assignment with hierarchy guard:

   Per user decision: use Server Actions instead of Route Handlers. Components call this function directly instead of fetching `/api/users/[id]/role`.

   NOTE: No Zod validation needed. Server Actions receive typed parameters directly from the calling component. TypeScript enforces the input shape at compile time.

   ```typescript
   "use server"

   import { auth } from "@/lib/auth"
   import { headers } from "next/headers"
   import { ROLE_LEVELS, type UserRole } from "@/types"

   const VALID_ROLES: UserRole[] = ["unassigned", "loanOfficer", "admin", "superAdmin"]

   export async function assignRole(input: { userId: string; role: UserRole }) {
     // Get current user session
     const session = await auth.api.getSession({
       headers: await headers(),
     })
     if (!session?.user) {
       return { error: "Unauthorized" }
     }

     const { userId, role: targetRole } = input

     // Validate input at runtime (TypeScript types are erased at runtime)
     if (!userId || typeof userId !== "string") {
       return { error: "User ID is required" }
     }
     if (!VALID_ROLES.includes(targetRole)) {
       return { error: "Invalid role" }
     }

     const actorRole = (session.user.role ?? "unassigned") as UserRole
     const actorLevel = ROLE_LEVELS[actorRole] ?? 0
     const targetLevel = ROLE_LEVELS[targetRole] ?? 0

     // AUTH-05: Cannot assign role at or above own level
     if (targetLevel >= actorLevel) {
       return { error: "Cannot assign role at or above your own level" }
     }

     // Only admin+ can assign roles at all
     if (actorLevel < ROLE_LEVELS.admin) {
       return { error: "Insufficient permissions to assign roles" }
     }

     try {
       await auth.api.setRole({
         body: { userId, role: targetRole },
         headers: await headers(),
       })
       return { success: true, role: targetRole }
     } catch (error) {
       return { error: "Failed to update role" }
     }
   }
   ```

   CRITICAL: The file is `proxy.ts`, NOT `middleware.ts`. Using `middleware.ts` will silently fail in Next.js 16.
   CRITICAL: Server Actions use `headers()` from `next/headers` to get the request headers for auth checks, NOT `request.headers`.
