**Step 3: Create optimistic add mutation.**
Replace the existing add handler with a useMutation:

```tsx
const [localTransactions, setLocalTransactions] = useState(initialTransactions)

const addMutation = useMutation({
  mutationFn: (input: { description: string; amount: string; categoryId: string; date: string }) =>
    recordExpenseAction(input),
  onMutate: async (input) => {
    const optimistic = {
      id: `optimistic-${Date.now()}`,
      description: input.description,
      amount: input.amount,
      categoryId: input.categoryId,
      date: input.date,
      type: "debit" as const,
      isOptimistic: true,
      createdAt: new Date().toISOString(),
      // Include any other fields the Transaction type requires with placeholder values
    }
    const previous = [...localTransactions]
    setLocalTransactions((prev) => [optimistic as any, ...prev])
    return { previous, optimisticId: optimistic.id }
  },
  onError: (_err, _input, context) => {
    if (context?.previous) setLocalTransactions(context.previous)
    toast.error("Failed to record expense")
  },
  onSuccess: (result, _input, context) => {
    if (result && "error" in result) {
      if (context?.previous) setLocalTransactions(context.previous)
      toast.error(result.error)
      return
    }
    // Replace optimistic item with real data from server
    if (result && "data" in result && context?.optimisticId) {
      setLocalTransactions((prev) =>
        prev.map((t) => (t.id === context.optimisticId ? result.data : t))
      )
    }
    toast.success("Expense recorded")
    // Close the add form/sheet
  },
})
```

**Step 4: Create optimistic delete mutation.**
```tsx
const deleteMutation = useMutation({
  mutationFn: (transactionId: string) => deleteExpenseAction(transactionId),
  onMutate: async (transactionId) => {
    const previous = [...localTransactions]
    setLocalTransactions((prev) => prev.filter((t) => t.id !== transactionId))
    return { previous }
  },
  onError: (_err, _id, context) => {
    if (context?.previous) setLocalTransactions(context.previous)
    toast.error("Failed to delete expense")
  },
  onSuccess: (result, _id) => {
    if (result && "error" in result) {
      // Rollback handled by refetching — but show toast
      toast.error(result.error)
      router.refresh()
      return
    }
    toast.success("Expense deleted")
  },
})
```

**Step 5: Update the submit handlers** to call `addMutation.mutate(input)` and `deleteMutation.mutate(transactionId)` instead of the previous startTransition pattern.

**Step 6: Update button states — CRITICAL for concurrent-mutation race condition prevention:**
- Add button: `disabled={addMutation.isPending}` — this disables the button from the moment `mutate()` is called until the mutation settles (success or error). This prevents double-submit race conditions where rapid clicks could create duplicate optimistic rows with stale snapshots (Pitfall 3 from RESEARCH.md).
- Delete button per row: `disabled={deleteMutation.isPending}` (or use `deleteMutation.variables === t.id` to disable only the row being deleted)
- Mark optimistic rows with reduced opacity: `className={t.isOptimistic ? "opacity-50" : ""}` on the table row

**Step 7: Remove the old `useTransition` for add/delete** (keep it for category management if categories also use startTransition). If the component has only one useTransition that was shared between add/delete/category operations, only remove the add/delete usage and keep category operations on useTransition.
