**3. Create 4 Route Handlers** — all follow the same pattern:

`src/app/api/reports/pnl/route.ts`:
```typescript
import { Effect } from "effect"
import { getPnlData } from "@/services/report.service"
import { generatePnlPdf } from "@/services/export/pdf.service"
import { generatePnlExcel } from "@/services/export/excel.service"

export async function GET(request: Request) {
  const { searchParams } = new URL(request.url)
  const format = searchParams.get("format") ?? "pdf"
  const period = searchParams.get("period") ?? getLastCompletedMonth()

  const data = await Effect.runPromise(getPnlData(period))

  if (format === "excel") {
    const buffer = await generatePnlExcel(data)
    return new Response(buffer, {
      headers: {
        "Content-Type": "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
        "Content-Disposition": `attachment; filename="pnl-${period}.xlsx"`,
      },
    })
  }

  const buffer = generatePnlPdf(data)
  return new Response(buffer, {
    headers: {
      "Content-Type": "application/pdf",
      "Content-Disposition": `attachment; filename="pnl-${period}.pdf"`,
    },
  })
}

function getLastCompletedMonth(): string {
  const now = new Date()
  const lastMonth = new Date(now.getFullYear(), now.getMonth() - 1, 1)
  return `${lastMonth.getFullYear()}-${String(lastMonth.getMonth() + 1).padStart(2, "0")}`
}
```

Repeat this pattern for:
- `/api/reports/portfolio/route.ts` — uses `getPortfolioData()`, `generatePortfolioPdf()`, `generatePortfolioExcel()`
- `/api/reports/balance-sheet/route.ts` — uses `getBalanceSheetData()`, `generateBalanceSheetPdf()`, `generateBalanceSheetExcel()`
- `/api/reports/transactions/route.ts` — uses `listTransactions({}, 1, 10000)` to get all, `generateTransactionsPdf()`, `generateTransactionsExcel()`

Each Route Handler returns Content-Disposition attachment with appropriate filename.
