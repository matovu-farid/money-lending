**2. Create `src/services/transaction.service.ts`:**

Exported functions:
- `recordExpense(input: CreateExpenseInput, actorId: string)`: Effect that inserts into `transactions` with `type: "debit"`, writes audit log. NOT inside a db.transaction since it's a single insert (audit log is separate call).
- `recordIncome(input: CreateIncomeInput, actorId: string)`: Effect that inserts into `transactions` with `type: "credit"`, writes audit log.
- `listTransactions(filters: TransactionLogFilters, page: number, pageSize: number)`: Effect that queries transactions with optional filters (type, categoryId, dateFrom, dateTo), joins with transactionCategories for category name, orders by transactionDate DESC, returns paginated results with total count.
- `deleteTransaction(id: string, actorId: string)`: Effect that deletes a transaction by ID. Writes audit log with before value.
- `getTransactionById(id: string)`: Effect that fetches a single transaction or returns TransactionNotFound.
- `autoPostInterestEarned(tx: DrizzleTransaction, params: { amount: string; loanId: string; paymentDate: string; actorId: string })`: Plain async function (NOT Effect â€” called inside db.transaction callback, same as writeAuditLog). Looks up "Interest Earned" category by name, inserts transaction with `type: "credit"`, `referenceType: "payment"`, `description: "Interest earned - loan {loanId}"`.
- `autoPostInterestExpense(tx: DrizzleTransaction, params: { amount: string; investmentId: string; repaymentDate: string; actorId: string })`: Plain async function. Looks up "Interest Payments" category by name, inserts transaction with `type: "debit"`, `referenceType: "creditor_repayment"`, `description: "Interest paid - investment {investmentId}"`.

The `autoPost*` functions must accept a Drizzle transaction object (`tx`) as their first parameter so they execute inside the same DB transaction as the parent operation (Pitfall 1 from RESEARCH.md).

```typescript
"use server"
import { Effect } from "effect"
import { recordExpense, deleteTransaction } from "@/services/transaction.service"
import { createCategory, deleteCategory } from "@/services/category.service"
import { revalidatePath } from "next/cache"
import { getSessionOrThrow } from "@/lib/auth/session-helpers" // existing helper

export async function recordExpenseAction(formData: FormData) {
  const session = await getSessionOrThrow()
  const input = {
    categoryId: formData.get("categoryId") as string,
    amount: formData.get("amount") as string,
    transactionDate: formData.get("transactionDate") as string,
    notes: formData.get("notes") as string | undefined,
  }
  await Effect.runPromise(recordExpense(input, session.user.id))
  revalidatePath("/expenses")
  revalidatePath("/transactions")
}

export async function deleteExpenseAction(id: string) {
  const session = await getSessionOrThrow()
  await Effect.runPromise(deleteTransaction(id, session.user.id))
  revalidatePath("/expenses")
  revalidatePath("/transactions")
}

export async function createExpenseCategoryAction(formData: FormData) {
  const session = await getSessionOrThrow()
  const input = { name: formData.get("name") as string, type: "expense" as const }
  await Effect.runPromise(createCategory(input, session.user.id))
  revalidatePath("/expenses")
}

export async function deleteExpenseCategoryAction(id: string) {
  const session = await getSessionOrThrow()
  await Effect.runPromise(deleteCategory(id, session.user.id))
  revalidatePath("/expenses")
}
```
