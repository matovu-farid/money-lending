1. Create `src/proxy.ts` (NOT middleware.ts — Next.js 16 breaking change):
   ```typescript
   import { NextResponse } from "next/server"
   import type { NextRequest } from "next/server"

   const AUTH_PAGES = ["/login", "/register", "/forgot-password"]
   const PUBLIC_PAGES = ["/pending-approval"]

   export async function proxy(request: NextRequest) {
     const sessionCookie = request.cookies.get("better-auth.session_token")
     const { pathname } = request.nextUrl
     const isAuthPage = AUTH_PAGES.some((p) => pathname.startsWith(p))

     // No session cookie — redirect to login unless on auth page
     if (!sessionCookie?.value) {
       if (isAuthPage) return NextResponse.next()
       return NextResponse.redirect(new URL("/login", request.url))
     }

     // Has session cookie — redirect away from auth pages to dashboard
     if (isAuthPage) {
       return NextResponse.redirect(new URL("/dashboard", request.url))
     }

     // Allow public pages (pending-approval) through
     if (PUBLIC_PAGES.some((p) => pathname.startsWith(p))) {
       return NextResponse.next()
     }

     // For all other routes, let the request through
     // The actual role check happens in the route handler (DAL pattern)
     // The proxy is optimistic — it only checks for cookie presence
     return NextResponse.next()
   }

   export const config = {
     matcher: ["/((?!api|_next/static|_next/image|favicon.ico).*)"],
   }
   ```

   NOTE: Per research open question #1, use cookie-only check in proxy for performance. The full `auth.api.getSession` with DB lookup happens in route handlers. The proxy is optimistic — it checks cookie existence, not validity.

2. Create `src/app/api/users/[id]/role/route.ts` — role assignment with hierarchy guard:
   ```typescript
   import { NextRequest } from "next/server"
   import { auth } from "@/lib/auth"
   import { z } from "zod"
   import { ROLE_LEVELS, type UserRole } from "@/types"

   const SetRoleSchema = z.object({
     role: z.enum(["unassigned", "loanOfficer", "admin", "superAdmin"]),
   })

   export async function POST(
     request: NextRequest,
     { params }: { params: Promise<{ id: string }> }
   ) {
     // Get current user session
     const session = await auth.api.getSession({ headers: request.headers })
     if (!session?.user) {
       return Response.json({ error: "Unauthorized" }, { status: 401 })
     }

     // Parse and validate request body
     const body = await request.json()
     const parsed = SetRoleSchema.safeParse(body)
     if (!parsed.success) {
       return Response.json({ error: parsed.error.flatten() }, { status: 400 })
     }

     const { role: targetRole } = parsed.data
     const actorRole = (session.user.role ?? "unassigned") as UserRole
     const actorLevel = ROLE_LEVELS[actorRole] ?? 0
     const targetLevel = ROLE_LEVELS[targetRole] ?? 0

     // AUTH-05: Cannot assign role at or above own level
     if (targetLevel >= actorLevel) {
       return Response.json(
         { error: "Cannot assign role at or above your own level" },
         { status: 403 }
       )
     }

     // Only admin+ can assign roles at all
     if (actorLevel < ROLE_LEVELS.admin) {
       return Response.json(
         { error: "Insufficient permissions to assign roles" },
         { status: 403 }
       )
     }

     const { id: userId } = await params

     try {
       await auth.api.setRole({
         body: { userId, role: targetRole },
         headers: request.headers,
       })
       return Response.json({ success: true, role: targetRole })
     } catch (error) {
       return Response.json(
         { error: "Failed to update role" },
         { status: 500 }
       )
     }
   }
   ```

   CRITICAL: The `params` are a Promise in Next.js 15+ — must `await params`.
   CRITICAL: The file is `proxy.ts`, NOT `middleware.ts`. Using `middleware.ts` will silently fail in Next.js 16.
