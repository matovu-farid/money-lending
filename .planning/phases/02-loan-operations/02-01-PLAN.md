    ```typescript
    export type PaymentAllocation = {
      interestPortion: string       // NUMERIC(15,2) string
      principalPortion: string
      principalBalanceBefore: string
      principalBalanceAfter: string
      loanFullyPaid: boolean
    }

    export function allocatePayment(params: {
      paymentAmount: string
      principalBalanceBefore: string
      monthlyRateDecimal: string
      daysElapsed: number
      minInterestDays: number
    }): PaymentAllocation {
      const { paymentAmount, principalBalanceBefore, monthlyRateDecimal, daysElapsed, minInterestDays } = params
      const payment = new BigNumber(paymentAmount)
      const interestOwed = calculateInterest(principalBalanceBefore, monthlyRateDecimal, daysElapsed, minInterestDays)

      if (payment.isLessThanOrEqualTo(interestOwed)) {
        return {
          interestPortion: formatAmount(payment),
          principalPortion: "0.00",
          principalBalanceBefore,
          principalBalanceAfter: principalBalanceBefore,
          loanFullyPaid: false,
        }
      }

      const principalPortion = payment.minus(interestOwed)
      const principalBalanceAfter = BigNumber.max(
        new BigNumber(principalBalanceBefore).minus(principalPortion),
        0
      )

      return {
        interestPortion: formatAmount(interestOwed),
        principalPortion: formatAmount(principalPortion),
        principalBalanceBefore,
        principalBalanceAfter: formatAmount(principalBalanceAfter),
        loanFullyPaid: principalBalanceAfter.isZero(),
      }
    }
    ```

    Also export `PaymentAllocation` type from `src/lib/interest/index.ts` barrel.

    **Create `src/services/payment.service.ts`** following the exact loan.service.ts Effect pattern:

    1. **`recordPayment(input: RecordPaymentInput, actorId: string)`** returns `Effect<Payment, LoanNotFound | DatabaseError>`:
       - Fetch loan by `input.loanId`. Fail with `LoanNotFound` if not found.
       - Inside `db.transaction(async (tx) => { ... })`:
         - Fetch all active payments (WHERE `deleted_at IS NULL`) for loan, ordered by `payment_date ASC, created_at ASC`
         - Determine `principalBalanceBefore`: if no prior payments, use `loan.principalAmount`; else use last payment's `principalBalanceAfter`
         - Determine `daysElapsed`: days between last payment's `payment_date` (or `loan.startDate` if first payment) and `input.paymentDate`
         - Determine effective `monthlyRateDecimal`: use `loan.interestRateOverride ?? loan.interestRate`
         - Determine effective `minInterestDays`: use `loan.minPeriodOverride ?? loan.minInterestDays`
         - Call `allocatePayment({ paymentAmount: input.amount, principalBalanceBefore, monthlyRateDecimal, daysElapsed, minInterestDays })`
         - Insert payment row with allocation results
         - If `allocation.loanFullyPaid` is true, update loan status to `"fully_paid"` using `eq(loans.id, input.loanId)`
         - If loan status is `"pending"`, update to `"active"` (Pitfall 6)
         - Call `await writeAuditLog(tx, { actorId, action: "payment.create", entityType: "payment", entityId: newPayment.id, beforeValue: null, afterValue: newPayment })`
         - Return the inserted payment row

    2. **`editPayment(input: EditPaymentInput, actorId: string)`** returns `Effect<Payment, PaymentNotFound | LoanNotFound | DatabaseError>`:
       - Fetch payment by `input.paymentId`. Fail with `PaymentNotFound` if not found or if `deletedAt` is not null.
       - Inside `db.transaction(async (tx) => { ... })`:
         - Save `beforeValue` snapshot of current payment
         - Update payment row with new `amount` and/or `paymentDate`, set `editReason = input.reason`, `updatedAt = new Date()`
         - Call `recalculateFromPayment(tx, payment.loanId, paymentIndex, allActivePayments)` to cascade
         - After recalculation, check if the final payment's `principalBalanceAfter` is zero -- update loan status to `"fully_paid"` if so, or back to `"active"` if previously `"fully_paid"` but now balance > 0
         - Call `await writeAuditLog(tx, { actorId, action: "payment.update", entityType: "payment", entityId: input.paymentId, beforeValue, afterValue: updatedPayment })` -- include `input.reason` in afterValue
         - Return updated payment

    3. **`deletePayment(input: DeletePaymentInput, actorId: string)`** returns `Effect<Payment, PaymentNotFound | LoanNotFound | DatabaseError>`:
       - Fetch payment by `input.paymentId`. Fail with `PaymentNotFound` if not found or already deleted.
       - Inside `db.transaction(async (tx) => { ... })`:
         - Set `deleted_at = new Date()`, `deleted_by = actorId`, `delete_reason = input.reason` on the payment row (soft delete, NEVER hard delete)
         - Call `recalculateFromPayment(tx, payment.loanId, paymentIndex, allActivePaymentsExcludingDeleted)` to cascade
         - After recalculation, update loan status based on final balance (same logic as editPayment)
         - Call `await writeAuditLog(tx, { actorId, action: "payment.delete", entityType: "payment", entityId: input.paymentId, beforeValue: payment, afterValue: { ...payment, deletedAt, deleteReason: input.reason } })`
         - Return soft-deleted payment

    4. **`recalculateFromPayment(tx, loanId, fromIndex, orderedPayments)`** (internal, not exported):
       - Starting from `fromIndex`, replay `allocatePayment()` for each subsequent payment using the `principalBalanceAfter` from the previous payment as `principalBalanceBefore`
       - For `daysElapsed` of each payment: `daysBetween(prevPayment.paymentDate ?? loan.startDate, currentPayment.paymentDate)`
       - Update each payment row in the same `tx` with new `interestPortion`, `principalPortion`, `principalBalanceBefore`, `principalBalanceAfter`

    5. **`getPaymentsForLoan(loanId: string)`** returns `Effect<Payment[], LoanNotFound | DatabaseError>`:
       - Fetch ALL payments (including soft-deleted) for the loan, ordered by `payment_date ASC, created_at ASC`
       - This is for UI display -- soft-deleted rows shown with strikethrough

    6. **Helper: `daysBetween(from: Date, to: Date): number`** -- compute calendar days between two timestamps. Use `Math.floor((to.getTime() - from.getTime()) / (1000 * 60 * 60 * 24))`. This is integer day-count (non-monetary), so `Math.floor` is acceptable.
